1 /**
  2  * Returns the index of an array based on an offset. Assumes 0-based index for array. 
  3  * Wraps around, so for example length 3, index 2, offset 2 would return 1.
  4  * @param  {int} length - The number of elements in the array
  5  * @param  {int} index - The starting index
  6  * @param  {int} offset - How many items to offset. Can be negative.
  7  * @returns {int} The index offset.
  8  */
  9 export function GetIndexOffset(length, index, offset){
 10 	// Assumes 0-based index for array
 11 	var newIndex = (index + offset) % length
 12 	
 13 	if (newIndex < 0) {
 14 		newIndex = length + newIndex;
 15 	}
 16 	return newIndex;
 17 };
 18 	
 19 /**
 20  * Returns an item from an array based on the provided item, and offset. Wraps around,
 21  * so with an arr ['A', 'B', 'C', 'D', 'E'], item 'B', offset -3, would return 'D'
 22  * @param  {Array} arr - The array to get offset item from.
 23  * @param  {Object} item - An element of the array.
 24  * @param  {int} offset - How many items to offset. Can be negative.
 25  * @returns {Object} An element of the input array.
 26  */
 27 export function GetArrayOffset(arr, item, offset){
 28 	var index = arr.indexOf(item);
 29 	var newIndex = this.GetIndexOffset(arr.length, index, offset);
 30 	return arr[newIndex];
 31 };
